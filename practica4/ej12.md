PairNat = $\forall$ X. (Nat -> Nat -> X) -> X

pairNat: Nat -> Nat -> PairNat
pairNat = \a: Nat. \b: Nat. $\Gamma$ X. \f: (Nat -> Nat -> X). f <X> a b

fstNat: PairNat -> Nat
fstNat = \p: PairNat. p <Nat> (\a: Nat. \b: Nat. a)

sndNat: PairNat -> Nat
sndNat = \p: PairNat. p <Nat> (\a: Nat. \b: Nat. b)
